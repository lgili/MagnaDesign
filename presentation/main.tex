\section{App overview}
% ============================================================

\begin{frame}{The problem that motivated this tool}
  \begin{columns}[T,onlytextwidth]
    \column{0.55\textwidth}
    PFC inductor engineering today involves:
    \begin{itemize}
      \item \textbf{Scattered spreadsheets} per topology
      \item Validation \textbf{only via FEMM} (specialist operator)
      \item Hard to iterate over $L$, $N$, core, wire without
            redoing everything
      \item EMC compliance (IEC 61000-3-2) is a separate
            calculation
      \item No \textbf{design trail} between projects
    \end{itemize}
    \column{0.45\textwidth}
    \app integrates:
    \begin{itemize}
      \item Analytical engine, calibrated against catalogues
      \item Automated FEA (FEMMT/ONELAB + FEMM legacy)
      \item Multi-tier optimiser (cascade)
      \item Datasheet-style exporter
      \item Single GUI for 6 topologies
    \end{itemize}
  \end{columns}
\end{frame}

\begin{frame}{Supported topologies}
  \begin{table}
    \centering\small
    \begin{tabular}{@{}lll@{}}
      \toprule
      \textbf{Topology} & \textbf{Typical range} & \textbf{Applications} \\
      \midrule
      Boost CCM & 200--2000 W & Server PSUs, industrial drives \\
      Buck CCM & 100--1500 W & Industrial DC--DC converters \\
      Interleaved Boost & 1--10 kW & High-power PFC, ripple cancel.\\
      Flyback (DCM/CCM) & 30--300 W & SMPS, adapters, auxiliaries \\
      Line reactor (3$\phi$) & 5--500 kW & VFD inputs, IEC 61000-3-2 \\
      Passive choke & 100 W--10 kW & 1$\phi$ passive PFC \\
      \bottomrule
    \end{tabular}
  \end{table}
  \medskip

  Each topology has its own analytical computation but shares the
  same engine for \textbf{losses, thermal, FEA and optimisation}.
\end{frame}

\section{Example 1 -- Boost PFC 1.5 kW}
% ============================================================

\begin{frame}{Example 1 --- universal-input boost PFC, 1.5 kW}
  \begin{columns}[T,onlytextwidth]
    \column{0.5\textwidth}
    \textbf{Specification}
    \begin{itemize}
      \item $V_\text{in} = 85$--$265$ V$_\text{rms}$
      \item $V_\text{out} = 400$ V (PFC bus)
      \item $P_\text{out} = 1500$ W
      \item $f_\text{sw} = 100$ kHz
      \item Ripple target: 30 \% of $I_\text{pk}$
    \end{itemize}

    \textbf{Market reference}
    \begin{itemize}
      \item TI UCC28019 application note
      \item Magnetics Kool-Mu 0077439A7 toroid
      \item AWG 16 solid, 55--70 turns
    \end{itemize}

    \column{0.5\textwidth}
    \textbf{Why this example?}
    \begin{itemize}
      \item \textbf{Most common} case in the market --- server
            PSU, telecom, industrial drives
      \item Power range where litz vs solid is still a real
            design decision
      \item Demonstrates \textbf{$\mu$ rolloff} under DC bias
            (Kool-Mu drops 50--70\% at peak)
      \item Exercises every FEA-dialog tab
    \end{itemize}
  \end{columns}
\end{frame}

\section{Example 2 -- 3$\phi$ line reactor, 22 kW}
% ============================================================

\begin{frame}{Example 2 --- 3$\phi$ line reactor for a VFD}
  \begin{columns}[T,onlytextwidth]
    \column{0.5\textwidth}
    \textbf{Specification}
    \begin{itemize}
      \item Mains: 400 V$_\text{LL}$, 60 Hz, 3-phase
      \item Load: 22 kW (motor drive)
      \item $I_\text{rms} \approx 32$ A per phase
      \item Target impedance: \textbf{3 \%}
      \item $L \approx 2.5$ mH per phase
    \end{itemize}

    \textbf{Reference}
    \begin{itemize}
      \item ABB MCB-32 / Schaffner FN3220
      \item M19 silicon-steel core
      \item $\sim$ 250 turns AWG 12 solid
      \item IEC 61000-3-2 Class A
    \end{itemize}

    \column{0.5\textwidth}
    \textbf{Why this example?}
    \begin{itemize}
      \item \textbf{Line-frequency} topology (60 Hz, no high-
            frequency switching)
      \item Triggers the \textbf{IEC 61000-3-2 harmonics}
            card
      \item N = 250 turns $\Rightarrow$ exercises the
            FEMMT $\to$ FEMM legacy auto-fallback
      \item Pass or fail EMC compliance? Visual answer in
            seconds
    \end{itemize}
  \end{columns}
\end{frame}

\section{Example 3 -- Flyback DCM, 65 W}
% ============================================================

\begin{frame}{Example 3 --- DCM flyback, 65 W}
  \begin{columns}[T,onlytextwidth]
    \column{0.5\textwidth}
    \textbf{Specification}
    \begin{itemize}
      \item $V_\text{in} = 85$--$265$ V$_\text{rms}$
      \item $V_\text{out} = 19$ V (laptop)
      \item $I_\text{out} = 3.4$ A $\Rightarrow$ 65 W
      \item $f_\text{sw} = 65$ kHz, DCM mode
      \item $L_\text{p} \approx 350~\mu$H
    \end{itemize}

    \textbf{Reference}
    \begin{itemize}
      \item TI UCC28911 EVM (laptop adapter)
      \item PQ20/16 N97 ferrite core
      \item $N_\text{p} = 42$, $N_\text{s} = 12$ ($n = 3.5$)
      \item Coupled inductor, 2 windings
    \end{itemize}

    \column{0.5\textwidth}
    \textbf{Why this example?}
    \begin{itemize}
      \item Coupled-pair --- entirely different from a
            single-winding choke
      \item Triggers the \textbf{2-winding render}
            (primary + secondary in distinct colours) on
            the geometry view
      \item Exercises the \textbf{leakage $L_\text{lk}$}
            calculation (Mohan / Erickson formula)
      \item DCM --- discontinuous triangular current
            waveform
    \end{itemize}
  \end{columns}
\end{frame}
